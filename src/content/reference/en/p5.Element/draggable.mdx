  <p>Turns p5.Element into a draggable item. If an argument is given, it will
  drag that p5.Element instead, ie. drag a entire GUI panel (parent container)
  with the panel's title bar.</p>

example:
  - |-

    <div><code>
    function setup() {
      createCanvas(100, 100);
      background(200);

      createDiv('Post-It')
        .position(5, 5)
        .size(75, 20)
        .style('font-size', '16px')
        .style('background', 'yellow')
        .style('color', '#000')
        .style('border', '1px solid #aaaa00')
        .style('padding', '5px')
        .draggable();
      // .style('cursor', 'help') // override cursor

      let gui = createDiv('')
        .position(5, 40)
        .size(85, 50)
        .style('font-size', '16px')
        .style('background', 'yellow')
        .style('z-index', '100')
        .style('border', '1px solid #00aaaa');

      createDiv('= PANEL =')
        .parent(gui)
        .style('background', 'cyan')
        .style('padding', '2px')
        .style('text-align', 'center')
        .draggable(gui);

      createSlider(0, 100, random(100))
        .style('cursor', 'pointer')
        .size(80, 5)
        .parent(gui);
    }
