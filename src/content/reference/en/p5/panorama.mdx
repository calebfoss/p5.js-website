description: >
  <p>The <code>panorama(img)</code> method transforms images containing

  360-degree content, such as maps or HDRIs, into immersive

  3D backgrounds that surround your scene. This is similar to calling

  <code>background(color)</code>; call <code>panorama(img)</code> before drawing
  your

  scene to create a 360-degree background from your image. It

  operates on the concept of sphere mapping, where the image is

  mapped onto an infinitely large sphere based on the angles of the

  camera.</p>

  <p>To enable 360-degree viewing, either use <code>orbitControl()</code> or try
  changing

  the orientation of the camera to see different parts of the background.</p>

    <code>
    let img;
    function preload() {
      img = loadImage('/assets/outdoor_spheremap.jpg');
    }
    function setup() {
      createCanvas(100 ,100 ,WEBGL);
    }
    function draw() {
      panorama(img);
      imageMode(CENTER);
      orbitControl();
      noStroke();
      push();
      imageLight(img);
      specularMaterial('green');
      shininess(200);
      metalness(100);
      sphere(25);
      pop();
    }
