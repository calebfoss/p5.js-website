  <p>A numeric variable that stores the width of the drawing canvas. Its

  default value is 100.</p>

  <p>Calling <a href="#/p5/createCanvas">createCanvas()</a> or

  <a href="#/p5/resizeCanvas">resizeCanvas()</a> changes the value of

  <code>width</code>. Calling <a href="#/p5/noCanvas">noCanvas()</a> sets its
  value to

  0.</p>

    <div>
    <code>
    function setup() {
      background(200);

      // Display the canvas' width.
      text(width, 42, 54);

      describe('The number 100 written in black on a gray square.');
    }
    </code>
    </div>

    }
    </code>
    </div>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // Display the canvas' width.
      text(width, 42, 54);

      describe('The number 100 written in black on a gray square. When the mouse is pressed, the square becomes a rectangle and the number becomes 50.');
    }

    // If the mouse is pressed, reisze
    // the canvas and display its new
    // width.
    function mousePressed() {
      if (mouseX > 0 && mouseX < width && mouseY > 0 && mouseY < height) {
        resizeCanvas(50, 100);
        background(200);
        text(width, 21, 54);
      }
    }
    </code>
    </div>
