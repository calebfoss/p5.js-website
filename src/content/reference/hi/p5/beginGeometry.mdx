  <p>एक नए <a href="/reference/p5/p5.Geometry">p5.Geometry</a> ऑब्जेक्ट में आकृतियाँ जोड़ना शुरू करता है।</p>

  <p> <code>beginGeometry()</code> और <a href="/reference/p5/endGeometry">endGeometry()</a> फ़ंक्शन <a href="/reference/p5/sphere">sphere()</a> जैसी सरल आकृतियों से जटिल 3D आकृतियाँ बनाने में सहायता करते हैं|
  
  <code>beginGeometry()</code> एक कस्टम <a href="/reference/p5/p5.Geometry">p5.Geometry</a> ऑब्जेक्ट में आकृतियाँ जोड़ना शुरू करता है और <a href="/reference/p5/EndGeometry">endGeometry()</a> उन्हें जोड़ना बंद कर देता है।</p>

  <p><code>beginGeometry()</code> और <a href="/reference/p5/endGeometry">endGeometry()</a> रेखाचित्रों को और अधिक उपयोगी बनाने में मदद कर सकते हैं।
  
  उदाहरण के लिए, यदि स्केच चलते समय कोई जटिल 3D आकार नहीं बदलता है, तो इसे <code>beginGeometry()</code> और <a href="/reference/p5/endGeometry">endGeometry() के साथ बनाया जा सकता है। </a>.

  <a href="/reference/p5/p5.Geometry">p5.Geometry</a> ऑब्जेक्ट को एक बार बनाना और फिर उसे खींचना अलग-अलग टुकड़ों को बार-बार खींचने की तुलना में तेज़ चलेगा।</p>

  <p>3D आकार बनाने के दूसरे तरीके के लिए <a href="/reference/p5/buildGeometry">buildGeometry()</a> देखें।</p>

  <p>ध्यान दें: <code>beginGeometry()</code> का उपयोग केवल WebGL मोड में किया जा सकता है।</p>

    <code>
    let shapes;

    function setup() {
      createCanvas(100, 100, WEBGL);
      makeShapes();
    }

    function makeShapes() {
      beginGeometry();
      scale(0.18);

      push();
      translate(100, -50);
      scale(0.5);
      rotateX(PI/4);
      cone();
      pop();
      cone();

      beginShape();
      vertex(-20, -50);
      quadraticVertex(
        -40, -70,
        0, -60
      );
      endShape();

      beginShape(TRIANGLE_STRIP);
      for (let y = 20; y <= 60; y += 10) {
        for (let x of [20, 60]) {
          vertex(x, y);
        }
      }
      endShape();

      beginShape();
      vertex(-100, -120);
      vertex(-120, -110);
      vertex(-105, -100);
      endShape();

      shapes = endGeometry();
    }

    function draw() {
      background(255);
      lights();
      orbitControl();
      model(shapes);
    }
    </code>
