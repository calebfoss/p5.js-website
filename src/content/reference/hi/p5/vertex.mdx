  <p>कस्टम आकार में एक शीर्ष जोड़ता है।</p>

  <p><code>vertex()</code> <a href="/reference/p5/beginShape">beginShape()</a> और <a href="/reference/p5/endShape">endShape()</a> फ़ंक्शन के बीच खींचे गए शीर्षों के निर्देशांक सेट करता है।</p>

  <p>पहले दो पैरामीटर, <code>x</code> और <code>y</code>, शीर्ष के x- और y-निर्देशांक सेट करते हैं।</p>

  <p>तीसरा पैरामीटर, <code>z</<p>कस्टम आकार में एक शीर्ष जोड़ता है। यह WebGL मोड में शीर्ष का z-निर्देशांक सेट करता है। डिफ़ॉल्ट रूप से, <code>z</code> 0 है।</p>

  <p>चौथा और पांचवां पैरामीटर, <code>u</code> और <code>v</code>, भी वैकल्पिक हैं। <a href="/reference/p5/endShape">endShape()</a> के साथ उपयोग करने पर वे शीर्ष की बनावट के लिए u- और v-निर्देशांक सेट करते हैं। डिफ़ॉल्ट रूप से, <code>u</code> और <code>v</code> दोनों 0 हैं।</p>

    // Click to change the number of sides.
    // In WebGL mode, custom shapes will only
    // display hollow fill sections when
    // all calls to vertex() use the same z-value.

    let sides = 3;
    let angle, px, py;

    function setup() {
      createCanvas(100, 100, WEBGL);
      setAttributes('antialias', true);
      fill(237, 34, 93);
      strokeWeight(3);
    }

    function draw() {
      background(200);
      rotateX(frameCount * 0.01);
      rotateZ(frameCount * 0.01);
      ngon(sides, 0, 0, 80);
    }

    function mouseClicked() {
      if (sides > 6) {
        sides = 3;
      } else {
        sides++;
      }
    }

    function ngon(n, x, y, d) {
      beginShape(TESS);
      for (let i = 0; i < n + 1; i++) {
        angle = TWO_PI / n * i;
        px = x + sin(angle) * d / 2;
        py = y - cos(angle) * d / 2;
        vertex(px, py, 0);
      }
      for (let i = 0; i < n + 1; i++) {
        angle = TWO_PI / n * i;
        px = x + sin(angle) * d / 4;
        py = y - cos(angle) * d / 4;
        vertex(px, py, 0);
      }
      endShape();
    }
    </code>
